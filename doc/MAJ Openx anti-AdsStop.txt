# Modification d'openx pour passer entre les filets des logiciels AdsStop

## Se placer dans le répertoire d'openx

chmod 777 -R www
chmod 777 -R var
cd www
mv delivery jedelivre 
mv images mesimages
cd jedelivre
mv ac.php ox-ac.php
mv afr.php ox-afr.php
mv ai.php ox-ai.php
mv ajs.php ox-ajs.php
mv al.php ox-al.php
mv alocal.php ox-alocal.php
mv apu.php ox-apu.php
mv avw.php ox-avw.php
mv axmlrpc.php ox-axmlrpc.php
mv ck.php ox-ck.php
mv fc.php ox-fc.php
mv ti.php ox-ti.php
mv tjs.php ox-tjs.php
mv tv.php ox-tv.php
mv fl.js ox-fl.js
cd ../..
